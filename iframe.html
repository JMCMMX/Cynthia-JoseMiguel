        function startCountdown(targetDate) {
            let targetTime = new Date(targetDate).getTime();

            let timer = setInterval(function () {
                let now = new Date().getTime();
                let timeLeft = targetTime - now;

                if (timeLeft <= 0) {
                    clearInterval(timer);
                    document.getElementById("countdown").innerHTML = "¡El evento ha comenzado!";
                    return;
                }

                let days = Math.ceil(timeLeft / (1000 * 60 * 60 * 24));

                let countdownElement = document.getElementById("countdown");
                countdownElement.innerHTML = `${days} días!`;

                // Añadir la animación de "pop"
                countdownElement.style.animation = "popEffect 0.5s ease-in-out";
                
                // Resetear animación después de 0.5s
                setTimeout(() => {
                    countdownElement.style.animation = "";
                }, 500);

            }, 1000);
        }

        // Configura la fecha objetivo: 6 de febrero de 2026
        startCountdown("2026-02-06 00:00:00");
    </script>
